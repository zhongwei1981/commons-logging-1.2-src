031/**
032 * A ClassLoader which sees only specified classes, and which can be
033 * set to do parent-first or child-first path lookup.
034 * <p>
035 * Note that this classloader is not "industrial strength"; users
036 * looking for such a class may wish to look at the Tomcat sourcecode
037 * instead. In particular, this class may not be threadsafe.
038 * <p>
039 * Note that the ClassLoader.getResources method isn't overloaded here.
040 * It would be nice to ensure that when child-first lookup is set the
041 * resources from the child are returned earlier in the list than the
042 * resources from the parent. However overriding this method isn't possible
043 * as the java 1.4 version of ClassLoader declares this method final
044 * (though the java 1.5 version has removed the final qualifier). As the
045 * ClassLoader javadoc doesn't specify the order in which resources
046 * are returned, it's valid to return the resources in any order (just
047 * untidy) so the inherited implementation is technically ok.
048 */
049
050public class PathableClassLoader extends URLClassLoader {
051    
052    private static final URL[] NO_URLS = new URL[0];
053    
054    /**
055     * A map of package-prefix to ClassLoader. Any class which is in
056     * this map is looked up via the specified classloader instead of
057     * the classpath associated with this classloader or its parents.
058     * <p>
059     * This is necessary in order for the rest of the world to communicate
060     * with classes loaded via a custom classloader. As an example, junit
061     * testcases which are loaded via a custom classloader needs to see
062     * the same junit classes as the code invoking the testcase, otherwise
063     * they can't pass result objects back. 
064     * <p>
065     * Normally, only a classloader created with a null parent needs to
066     * have any lookasides defined.
067     */
068    private HashMap lookasides = null;
069
070    /**
071     * See setParentFirst.
072     */
073    private boolean parentFirst = true;
074
075    /**
076     * Constructor.
077     * <p>
078     * Often, null is passed as the parent, ie the parent of the new
079     * instance is the bootloader. This ensures that the classpath is
080     * totally clean; nothing but the standard java library will be
081     * present.
082     * <p>
083     * When using a null parent classloader with a junit testcase, it *is*
084     * necessary for the junit library to also be visible. In this case, it
085     * is recommended that the following code be used:
086     * <pre>
087     * pathableLoader.useExplicitLoader(
088     *   "junit.",
089     *   junit.framework.Test.class.getClassLoader());
090     * </pre>
091     * Note that this works regardless of whether junit is on the system
092     * classpath, or whether it has been loaded by some test framework that
093     * creates its own classloader to run unit tests in (eg maven2's
094     * Surefire plugin).
095     */
096    public PathableClassLoader(ClassLoader parent) {
097        super(NO_URLS, parent);
098    }
099    
100    /**
101     * Allow caller to explicitly add paths. Generally this not a good idea;
102     * use addLogicalLib instead, then define the location for that logical
103     * library in the build.xml file.
104     */
105    public void addURL(URL url) {
106        super.addURL(url);
107    }
108
109    /**
110     * Specify whether this classloader should ask the parent classloader
111     * to resolve a class first, before trying to resolve it via its own
112     * classpath.
113     * <p> 
114     * Checking with the parent first is the normal approach for java, but
115     * components within containers such as servlet engines can use 
116     * child-first lookup instead, to allow the components to override libs
117     * which are visible in shared classloaders provided by the container.
118     * <p>
119     * Note that the method getResources always behaves as if parentFirst=true,
120     * because of limitations in java 1.4; see the javadoc for method
121     * getResourcesInOrder for details.
122     * <p>
123     * This value defaults to true.
124     */
125    public void setParentFirst(boolean state) {
126        parentFirst = state;
127    }
128
129    /**
130     * For classes with the specified prefix, get them from the system
131     * classpath <i>which is active at the point this method is called</i>.
132     * <p>
133     * This method is just a shortcut for
134     * <pre>
135     * useExplicitLoader(prefix, ClassLoader.getSystemClassLoader());
136     * </pre>
137     * <p>
138     * Of course, this assumes that the classes of interest are already
139     * in the classpath of the system classloader.
140     */
141    public void useSystemLoader(String prefix) {
142        useExplicitLoader(prefix, ClassLoader.getSystemClassLoader());
143        
144    }
145
146    /**
147     * Specify a classloader to use for specific java packages.
148     * <p>
149     * The specified classloader is normally a loader that is NOT
150     * an ancestor of this classloader. In particular, this loader
151     * may have the bootloader as its parent, but be configured to 
152     * see specific other classes (eg the junit library loaded
153     * via the system classloader).
154     * <p>
155     * The differences between using this method, and using
156     * addLogicalLib are:
157     * <ul>
158     * <li>If code calls getClassLoader on a class loaded via
159     * "lookaside", then traces up its inheritance chain, it
160     * will see the "real" classloaders. When the class is remapped
161     * into this classloader via addLogicalLib, the classloader
162     * chain seen is this object plus ancestors.
163     * <li>If two different jars contain classes in the same
164     * package, then it is not possible to load both jars into
165     * the same "lookaside" classloader (eg the system classloader)
166     * then map one of those subsets from here. Of course they could
167     * be loaded into two different "lookaside" classloaders and
168     * then a prefix used to map from here to one of those classloaders.
169     * </ul>
170     */
171    public void useExplicitLoader(String prefix, ClassLoader loader) {
172        if (lookasides == null) {
173            lookasides = new HashMap();
174        }
175        lookasides.put(prefix, loader);
176    }
177
178    /**
179     * Specify a collection of logical libraries. See addLogicalLib.
180     */
181    public void addLogicalLib(String[] logicalLibs) {
182        for(int i=0; i<logicalLibs.length; ++i) {
183            addLogicalLib(logicalLibs[i]);
184        }
185    }
186
187    /**
188     * Specify a logical library to be included in the classpath used to
189     * locate classes. 
190     * <p>
191     * The specified lib name is used as a key into the system properties;
192     * there is expected to be a system property defined with that name
193     * whose value is a url that indicates where that logical library can
194     * be found. Typically this is the name of a jar file, or a directory
195     * containing class files.
196     * <p>
197     * If there is no system property, but the classloader that loaded
198     * this class is a URLClassLoader then the set of URLs that the
199     * classloader uses for its classpath is scanned; any jar in the
200     * URL set whose name starts with the specified string is added to
201     * the classpath managed by this instance. 
202     * <p>
203     * Using logical library names allows the calling code to specify its
204     * desired classpath without knowing the exact location of the necessary
205     * classes. 
206     */
207    public void addLogicalLib(String logicalLib) {
208        // first, check the system properties
209        String filename = System.getProperty(logicalLib);
210        if (filename != null) {
211            try {
212                URL libUrl = new File(filename).toURL();
213                addURL(libUrl);
214                return;
215            } catch(java.net.MalformedURLException e) {
216                throw new UnknownError(
217                    "Invalid file [" + filename + "] for logical lib [" + logicalLib + "]");
218            }
219        }
220
221        // now check the classpath for a similar-named lib
222        URL libUrl = libFromClasspath(logicalLib);
223        if (libUrl != null) {
224            addURL(libUrl);
225            return;
226        }
227
228        // lib not found
229        throw new UnknownError(
230            "Logical lib [" + logicalLib + "] is not defined"
231            + " as a System property.");
232    }
233
234    /**
235     * If the classloader that loaded this class has this logical lib in its
236     * path, then return the matching URL otherwise return null.
237     * <p>
238     * This only works when the classloader loading this class is an instance
239     * of URLClassLoader and thus has a getURLs method that returns the classpath
240     * it uses when loading classes. However in practice, the vast majority of the
241     * time this type is the classloader used.
242     * <p>
243     * The classpath of the classloader for this instance is scanned, and any
244     * jarfile in the path whose name starts with the logicalLib string is
245     * considered a match. For example, passing "foo" will match a url
246     * of <code>file:///some/where/foo-2.7.jar</code>.
247     * <p>
248     * When multiple classpath entries match the specified logicalLib string,
249     * the one with the shortest filename component is returned. This means that
250     * if "foo-1.1.jar" and "foobar-1.1.jar" are in the path, then a logicalLib
251     * name of "foo" will match the first entry above.
252     */
253    private URL libFromClasspath(String logicalLib) {
254        ClassLoader cl = this.getClass().getClassLoader();
255        if (cl instanceof URLClassLoader == false) {
256            return null;
257        }
258        
259        URLClassLoader ucl = (URLClassLoader) cl;
260        URL[] path = ucl.getURLs();
261        URL shortestMatch = null;
262        int shortestMatchLen = Integer.MAX_VALUE;
263        for(int i=0; i<path.length; ++i) {
264            URL u = path[i];
265            
266            // extract the filename bit on the end of the url
267            String filename = u.toString();
268            if (!filename.endsWith(".jar")) {
269                // not a jarfile, ignore it
270                continue;
271            }
272
273            int lastSlash = filename.lastIndexOf('/');
274            if (lastSlash >= 0) {
275                filename = filename.substring(lastSlash+1);
276            }
277            
278            if (filename.startsWith(logicalLib)) {
279                // ok, this is a candidate
280                if (filename.length() < shortestMatchLen) {
281                    shortestMatch = u;
282                    shortestMatchLen = filename.length();
283                }
284            }
285        }
286        
287        return shortestMatch;
288    }

290    /**
291     * Override ClassLoader method.
292     * <p>
293     * For each explicitly mapped package prefix, if the name matches the 
294     * prefix associated with that entry then attempt to load the class via 
295     * that entries' classloader.
296     */
297    protected Class loadClass(String name, boolean resolve) 
298    throws ClassNotFoundException {
299        // just for performance, check java and javax
300        if (name.startsWith("java.") || name.startsWith("javax.")) {
301            return super.loadClass(name, resolve);
302        }
303
304        if (lookasides != null) {
305            for(Iterator i = lookasides.entrySet().iterator(); i.hasNext(); ) {
306                Map.Entry entry = (Map.Entry) i.next();
307                String prefix = (String) entry.getKey();
308                if (name.startsWith(prefix) == true) {
309                    ClassLoader loader = (ClassLoader) entry.getValue();
310                    Class clazz = Class.forName(name, resolve, loader);
311                    return clazz;
312                }
313            }
314        }
315        
316        if (parentFirst) {
317            return super.loadClass(name, resolve);
318        } else {
319            // Implement child-first. 
320            //
321            // It appears that the findClass method doesn't check whether the
322            // class has already been loaded. This seems odd to me, but without
323            // first checking via findLoadedClass we can get java.lang.LinkageError
324            // with message "duplicate class definition" which isn't good.
325            
326            try {
327                Class clazz = findLoadedClass(name);
328                if (clazz == null) {
329                    clazz = super.findClass(name);
330                }
331                if (resolve) {
332                    resolveClass(clazz);
333                }
334                return clazz;
335            } catch(ClassNotFoundException e) {
336                return super.loadClass(name, resolve);
337            }
338        }
339    }

341    /**
342     * Same as parent class method except that when parentFirst is false
343     * the resource is looked for in the local classpath before the parent
344     * loader is consulted.
345     */
346    public URL getResource(String name) {
347        if (parentFirst) {
348            return super.getResource(name);
349        } else {
350            URL local = super.findResource(name);
351            if (local != null) {
352                return local;
353            }
354            return super.getResource(name);
355        }
356    }
357    
358    /**
359     * Emulate a proper implementation of getResources which respects the
360     * setting for parentFirst.
361     * <p>
362     * Note that it's not possible to override the inherited getResources, as
363     * it's declared final in java1.4 (thought that's been removed for 1.5).
364     * The inherited implementation always behaves as if parentFirst=true.
365     */
366    public Enumeration getResourcesInOrder(String name) throws IOException {
367        if (parentFirst) {
368            return super.getResources(name);
369        } else {
370            Enumeration localUrls = super.findResources(name);
371            
372            ClassLoader parent = getParent();
373            if (parent == null) {
374                // Alas, there is no method to get matching resources
375                // from a null (BOOT) parent classloader. Calling
376                // ClassLoader.getSystemClassLoader isn't right. Maybe
377                // calling Class.class.getResources(name) would do?
378                //
379                // However for the purposes of unit tests, we can
380                // simply assume that no relevant resources are
381                // loadable from the parent; unit tests will never be
382                // putting any of their resources in a "boot" classloader
383                // path!
384                return localUrls;
385            }
386            Enumeration parentUrls = parent.getResources(name);
387
388            ArrayList localItems = toList(localUrls);
389            ArrayList parentItems = toList(parentUrls);
390            localItems.addAll(parentItems);
391            return Collections.enumeration(localItems);
392        }
393    }
394    
395    /**
396     * 
397     * Clean implementation of list function of 
398     * {@link java.utils.Collection} added in JDK 1.4 
399     * @param en <code>Enumeration</code>, possibly null
400     * @return <code>ArrayList</code> containing the enumerated
401     * elements in the enumerated order, not null
402     */
403    private ArrayList toList(Enumeration en) {
404        ArrayList results = new ArrayList();
405        if (en != null) {
406            while (en.hasMoreElements()){
407                Object element = en.nextElement();
408                results.add(element);
409            }
410        }
411        return results;
412    }

414    /**
415     * Same as parent class method except that when parentFirst is false
416     * the resource is looked for in the local classpath before the parent
417     * loader is consulted.
418     */
419    public InputStream getResourceAsStream(String name) {
420        if (parentFirst) {
421            return super.getResourceAsStream(name);
422        } else {
423            URL local = super.findResource(name);
424            if (local != null) {
425                try {
426                    return local.openStream();
427                } catch(IOException e) {
428                    // TODO: check if this is right or whether we should
429                    // fall back to trying parent. The javadoc doesn't say...
430                    return null;
431                }
432            }
433            return super.getResourceAsStream(name);
434        }
435    }
